library(dplyr)
library(openxlsx)
library(glue)
ktable <- function(grade,min,max,ncol){
  kanji <- read.xlsx(xlsxFile = "joyo-kanji-review.xlsx", sheet = "Joyo-Kanji")
  kanji2 <- kanji[c(1,2,6,8,9,11,12)]
  kanji3 <- kanji2 %>% filter(Grade==g) %>% arrange(Grade,Kanji)
  kanji4 <- kanji3[c(2,4,5,6,7)] 
  n <- max - min + 1 
  kanji5 <- kanji4 %>% head(max) %>% 
    glue_data("\\kk{{{Kanji}}}{{{Bedeutung}}}{{{R1}}}{{{R2}}}{{{Meaning}}}", .na = "") 
  k5 <- kanji5[c(min:max)]
  nr <- ncol - n + (ncol*floor((n-1)/ncol))
  nal <- rep("",nr)
  k5 <- append(k5, nal)
  matrix(k5, ncol = ncol, byrow = TRUE)
}
ktable(g,min,max,ncol)
\end{minted}
